Ao criar uma conta, você concorda com os termos de uso.
const form = document.getElementById('form-register'); const btn = document.getElementById('btn-register'); const error = document.getElementById('error'); const success = document.getElementById('success'); form.addEventListener('submit', async (e) => { e.preventDefault(); error.style.display = 'none'; success.style.display = 'none'; // Validações básicas const nome = document.getElementById('nome').value.trim(); const email = document.getElementById('email').value.trim(); const senha = document.getElementById('senha').value; const senhaConfirm = document.getElementById('senha_confirm').value; if (nome.length < 3) { error.textContent = 'Nome deve ter no mínimo 3 caracteres'; error.style.display = 'block'; return; } if (senha !== senhaConfirm) { error.textContent = 'As senhas não coincidem'; error.style.display = 'block'; return; } if (senha.length < 8) { error.textContent = 'A senha deve ter no mínimo 8 caracteres'; error.style.display = 'block'; return; } // Validação de complexidade da senha const hasUpper = /[A-Z]/.test(senha); const hasLower = /[a-z]/.test(senha); const hasNumber = /[0-9]/.test(senha); const hasSpecial = /[!@#$%^&*(),.?":{}|<>]/.test(senha); if (!hasUpper || !hasLower || !hasNumber || !hasSpecial) { error.textContent = 'Senha deve conter maiúscula, minúscula, número e caractere especial'; error.style.display = 'block'; return; } btn.disabled = true; btn.textContent = 'Criando conta...'; try { const resp = await fetch('/auth/register', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ nome, email, senha }) }); const data = await resp.json(); if (!resp.ok) { throw new Error(data.detail || 'Erro ao criar conta'); } success.textContent = '✅ Conta criada com sucesso! Redirecionando...'; success.style.display = 'block'; // Aguardar 1.5s e redirecionar para login setTimeout(() => { window.location.href = '/login'; }, 1500); } catch (err) { error.textContent = err.message || 'Falha ao criar conta'; error.style.display = 'block'; btn.disabled = false; btn.textContent = 'Criar conta'; } }); // Validação em tempo real da confirmação de senha document.getElementById('senha_confirm').addEventListener('input', (e) => { const senha = document.getElementById('senha').value; const senhaConfirm = e.target.value; if (senhaConfirm && senha !== senhaConfirm) { e.target.style.borderColor = 'rgba(239,68,68,0.5)'; } else { e.target.style.borderColor = 'rgba(148,163,184,0.2)'; } });
